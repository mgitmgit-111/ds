1. Express CRUD Operations with REST API

CRUD (Create, Read, Update, Delete) operations allow you to manage student data through REST API endpoints. This example uses Express.js with MySQL database.

Database Connection

db.js 
const mysql = require('mysql2/promise');
const mysqlPool = mysql.createPool({
    host: 'localhost',
    user: 'root',
    password: '', // Update if necessary
    database: 'nodejsdb'
});
module.exports = mysqlPool;

Express App Entry Point

index.js 
const express = require('express');
const app = express();
const db = require('./db');
const studentRoutes = require('./routes/student.routes');
const bodyParser = require('body-parser');

app.use(bodyParser.json());
app.use('/students', studentRoutes);

db.query('select 1')
    .then(() => {
        console.log('Database Connection Success..');
        app.listen(3000, () => {
            console.log('Server listening to port no 3000');
        });
    })
    .catch(err => console.log('DB Connection Failed', err));

Student Controller

controllers/student.controller.js 
const db = require('../db');

// CREATE student
exports.createStudent = async (req, res) => {
    const { name, age } = req.body;
    await db.query('INSERT INTO students (name, age) VALUES (?, ?)', [name, age]);
    res.json({ message: "Student created" });
};

// READ all students
exports.getStudents = async (req, res) => {
    const [rows] = await db.query('SELECT * FROM students');
    res.json(rows);
};

// UPDATE student
exports.updateStudent = async (req, res) => {
    const { id } = req.params;
    const { name, age } = req.body;
    await db.query('UPDATE students SET name = ?, age = ? WHERE id = ?', [name, age, id]);
    res.json({ message: "Student updated" });
};

// DELETE student
exports.deleteStudent = async (req, res) => {
    const { id } = req.params;
    await db.query('DELETE FROM students WHERE id = ?', [id]);
    res.json({ message: "Student deleted" });
};

Student Routes

routes/student.routes.js 
const express = require('express');
const router = express.Router();
const studentCtrl = require('../controllers/student.controller');

router.post('/', studentCtrl.createStudent); // Create
router.get('/', studentCtrl.getStudents); // Read
router.put('/:id', studentCtrl.updateStudent); // Update
router.delete('/:id', studentCtrl.deleteStudent); // Delete

module.exports = router;

3. Session Tracking: Cookies and HTTP Sessions

Session tracking maintains user transactional history across multiple requests. Two main mechanisms: Cookies and HTTP Sessions.

Cookies Example

cookies.js 
const express = require('express');
const cookieParser = require('cookie-parser');
const app = express();

app.use(cookieParser());

app.get('/setcookie', (req, res) => {
    res.cookie('username', 'student123', { maxAge: 60000 }); // Cookie for 1 minute
    res.send('Cookie set!');
});

app.get('/getcookie', (req, res) => {
    const user = req.cookies['username'];
    res.send('Cookie value: ' + user);
});

app.listen(3000, () => {
    console.log('Server on 3000');
});

HTTP Session Example - Setup

index.js 
const express = require('express');
const bodyParser = require('body-parser');
const cookieParser = require('cookie-parser');
const handlers = require('./handlers');
const app = express();

app.use(bodyParser.json());
app.use(cookieParser());

app.post('/signin', handlers.signinHandler);
app.get('/welcome', handlers.welcomeHandler);
app.get('/logout', handlers.logoutHandler);

app.listen(3000, () => {
  console.log('Server started on port no 3000');
});

HTTP Session Example - Handler

handlers.js 
const uuid = require('uuid');
const users = { user1: 'password1', user2: 'password2' };

class Session {
  constructor(username, expiresAt) {
    this.username = username;
    this.expiresAt = expiresAt;
  }

  isExpired() {
    return this.expiresAt < new Date();
  }
}

const sessions = {};

// Signin Handler - creates a session and sends a cookie
exports.signinHandler = (req, res) => {
  const { username, password } = req.body;
  if (!username || users[username] !== password) {
    res.status(401).end();
    return;
  }

  const sessionToken = uuid.v4();
  const expiresAt = new Date(Date.now() + 120 * 1000); // 2 minutes
  sessions[sessionToken] = new Session(username, expiresAt);
  res.cookie('sessiontoken', sessionToken, { expires: expiresAt });
  res.end();
};

// Welcome Handler - checks session from cookie
exports.welcomeHandler = (req, res) => {
  const sessionToken = req.cookies['sessiontoken'];
  const userSession = sessions[sessionToken];

  if (!sessionToken || !userSession || userSession.isExpired()) {
    res.status(401).end();
    return;
  }
  res.send('Welcome ' + userSession.username).end();
};

// Logout Handler - clears session and cookie
exports.logoutHandler = (req, res) => {
  const sessionToken = req.cookies['sessiontoken'];
  delete sessions[sessionToken];
  res.cookie('sessiontoken', '', { expires: new Date() });
  res.end();
};

Session vs Cookies Comparison

Aspect Cookies HTTP Sessions
Data Storage Client browser Server
Security Less secure (visible) More secure
Size Limit ~4KB No limit
Persistence Can be persistent Session-based
Expiry Manual setting Configurable

4. JWT Authorization for Protected Endpoints

JWT (JSON Web Token) is a secure way to transmit information between parties. It's encoded as a token that clients send with requests to access protected resources.

JWT Setup

jwt-server.js 
const express = require('express');
const jwt = require('jsonwebtoken');
const app = express();

// Protected endpoint: Only accessible with valid JWT
app.post('/api/posts', verifyToken, (req, res) => {
  jwt.verify(req.token, 'mgitetnodejslab', (err, authData) => {
    if (err) {
      res.sendStatus(403); // Forbidden
    } else {
      res.json({ message: 'User Authenticated Successfully....!', authData });
    }
  });
});

// Endpoint to generate JWT for user
app.post('/api/getToken', (req, res) => {
  const user = { id: 1, fullname: 'Mallela Srikanth', email: 'mallelasrikanth@csemgit.ac.in' };
  jwt.sign(user, 'mgitetnodejslab', { expiresIn: '30s' }, (err, token) => {
    res.json({ token });
  });
});

// Middleware: Verify JWT Token from header
function verifyToken(req, res, next) {
  const bearerHeader = req.headers['authorization'];
  if (typeof bearerHeader !== 'undefined') {
    req.token = bearerHeader.split(' ')[1];
    next();
  } else {
    res.sendStatus(403);
  }
}

app.listen(3000, () => {
  console.log('Server listening to port no 3000');
});

How JWT Works

  1. Request Token: Call /api/getToken → Get JWT token
  2. Send Protected Request: Include token in Authorization header: Bearer <token>
  3. Verify Token: Server checks token validity
  4. Access Granted: If valid, user can access protected resource

JWT Structure

A JWT token consists of three parts separated by dots:

  • Header: Specifies token type (JWT) and hashing algorithm
  • Payload: Contains user data (claims)
  • Signature: Ensures token integrity
